# Como identificar o uuid dos tenants
docker compose exec api python -c "from src.database import engine; from sqlalchemy import text; conn = engine.connect(); result = conn.execute(text('SELECT id, nome_fantasia FROM tenants LIMIT 1')); print('\n👉 SEU TENANT ID REAL É:', [dict(r._mapping) for r in result], '\n'); conn.close()"
